00078 /*
00079  * Compute a POSIX 1003.2 checksum.  This routine has been broken out so that
00080  * other programs can use it.  It takes a file descriptor to read from and
00081  * locations to store the crc and the number of bytes read.  It returns 0 on
00082  * success and 1 on failure.  Errno is set on failure.
00083  */
00084 
00085 
00086 EXPORT_C int crc(register int fd, u_long *cval, u_long *clen)
00087 {
00088         u_long crc_total = (u_long)~0;  /* The crc over a number of files*/
00089         register u_char *p;
00090         register int nr;
00091         register u_long crc, len;
00092 //      u_char buf[16 * 1024]; // too large for stack
00093         u_char buf[3 * 1024];
00094 
00095 #define COMPUTE(var, ch)        (var) = (var) << 8 ^ crctab[(var) >> 24 ^ (ch)]
00096 
00097         crc = len = 0;
00098         crc_total = ~crc_total;
00099         while ((nr = read(fd, (char*)buf, sizeof(buf))) > 0)
00100                 for (len += nr, p = buf; nr--; ++p) {
00101                         COMPUTE(crc, *p);
00102                         COMPUTE(crc_total, *p);
00103                 }
00104         if (nr < 0)
00105                 return (1);
00106 
00107         *clen = len;
00108 
00109         /* Include the length of the file. */
00110         for (; len != 0; len >>= 8) {
00111                 COMPUTE(crc, len & 0xff);
00112                 COMPUTE(crc_total, len & 0xff);
00113         }
00114 
00115         *cval = ~crc;
00116         crc_total = ~crc_total;
00117         return (0);
00118 }
