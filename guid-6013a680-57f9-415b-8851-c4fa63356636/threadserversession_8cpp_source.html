00074                         break;
00075                 
00076                 case EThreadServerOpenDriver:
00077                         aMessage.Complete(iServer.OpenLogicalChannel());

00081                         r = iAsyncRequestHandler->ProcessRequest(aMessage);

00083                                 aMessage.Complete(r);

00085                         
00086                 case EThreadServerSendDataCancel: //cancel async
00087                         iAsyncRequestHandler->Cancel(); 
00088                         aMessage.Complete(KErrNone); 
00089                         break;
00090                         
00091                 case EThreadServerUnloadDeviceDriver:
00092                         aMessage.Complete(iServer.UnloadDevice());
00093                         break;
00094

00096                         User::Leave(KErrNotSupported);
