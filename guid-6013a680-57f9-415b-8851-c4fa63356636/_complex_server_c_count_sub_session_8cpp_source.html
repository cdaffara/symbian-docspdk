00048       
00049       // Max length set to the value of "deslen", but current length is zero
00050     buffer.CreateL(deslen);
00051       
00052       // Do the right cleanup if anything subsequently goes wrong
00053     buffer.CleanupClosePushL();
00054     
00055       // Copy the client's descriptor data into our buffer.
00056     aMessage.ReadL(0,buffer,0);
00057     
00058       // Now do a validation to make sure that the string only has digits
00059     if (buffer.Length() == 0)
00060         {
00061         User::Leave(ENonNumericString);
00062         }

00074                 User::Leave(ENonNumericString);
00075             }
