AuthAttributes ::= SET SIZE (1..MAX) OF Attribute UnauthAttributes ::= SET SIZE (1..MAX) OF Attribute

RecipientInfos ::= SET OF RecipientInfo

MessageAuthenticationCode ::= OCTET STRING
